%Plot Gaussian Function
%s is sigma
%m is mean
%d is gaussian distribution(x,mu,s)
mu = 5;
x = 0:1:10;
s1 = 0.5;
s2 = 1.0;
s3 = 1.5;
p1 = -0.5 * ((x - mu)/s1) .^ 2;
p2 = (s1 * sqrt(2*pi));
d1 = exp(p1) ./ p2; %Gaussian s1
p3 = -0.5 * ((x - mu)/s2) .^ 2;
p4 = (s2 * sqrt(2*pi));
d2 = exp(p3) ./ p4; %Gaussian s2
p5 = -0.5 * ((x - mu)/s3) .^ 2;
p6 = (s3 * sqrt(2*pi));
d3 = exp(p5) ./ p6; %Gaussian s3

plot(x,d1,"-r",x,d2,"--g",x,d3,":b") %Plot on same graph
xlabel("x")
ylabel("d(x-5,s)")
title("Gaussian Plot")
legend("d1","d2","d3")
